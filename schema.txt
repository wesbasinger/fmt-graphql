type Hours {
	_id: String!
	worker: String!
	session: Session!
	comment: String!
	datestamp: String!
	timeIn: Float!
	timeOut: Float
	cast: Cast!
}

type Session {
	slug: String!
	show: String!
	cast: [Cast]!
}

type Cast {
	_id: String!
	firstName: String!
	lastName: String!
	sessions: [Session]!
	hours: [Hours]!
}
